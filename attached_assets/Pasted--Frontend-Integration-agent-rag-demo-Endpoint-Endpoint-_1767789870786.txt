 Frontend Integration: /agent-rag-demo Endpoint

  Endpoint Details

  URL: https://api.airthreads.ai:5001/agent-rag-demo
  Method: POST
  Content-Type: application/json
  Response: Server-Sent Events (SSE) stream

  Request Format

  {
    "query": "show me coffee chats"
  }

  Response Stream Format

  Event 1: Tokens (streaming)
  data: {"token":"You"}
  data: {"token":" have"}
  data: {"token":" 5"}
  ...

  Event 2: Complete (with meeting cards)
  event: complete
  data: {
    "status": "complete",
    "meeting_emails": [
      {
        "id": "test_msg_0001",
        "subject": "Coffee chat next week?",
        "snippet": "This is test email #1...",
        "date": "2025-12-25T10:30:00",
        "from": {
          "email": "john@acme.com",
          "name": "John Smith"
        },
        "eventRelated": true,
        "eventType": "coffee_chat",
        "confidence": 0.85,
        "context": "Coffee chat invitation"
      },
      ...
    ],
    "timing": {
      "first_token": 0.4,
      "total": 1.3,
      "tokens": 150
    },
    "metadata": {
      "query": "show me coffee chats",
      "result_count": 5,
      "meetings_count": 5
    }
  }

  React/TypeScript Implementation

  import { useState } from 'react';

  interface Message {
    role: 'user' | 'assistant';
    content: string;
    meetingEmails?: MeetingEmail[];
  }

  interface MeetingEmail {
    id: string;
    subject: string;
    snippet: string;
    date: string;
    from: {
      email: string;
      name: string;
    };
    eventRelated: boolean;
    eventType: string;
    confidence: number;
    context: string;
  }

  export function RAGDemoChat() {
    const [messages, setMessages] = useState<Message[]>([]);
    const [input, setInput] = useState('');
    const [isStreaming, setIsStreaming] = useState(false);

    const sendMessage = async () => {
      if (!input.trim() || isStreaming) return;

      // Add user message
      const userMessage: Message = { role: 'user', content: input };
      setMessages(prev => [...prev, userMessage]);
      setInput('');
      setIsStreaming(true);

      // Create assistant message
      let assistantMessage: Message = { role: 'assistant', content: '' };
      setMessages(prev => [...prev, assistantMessage]);

      try {
        const response = await fetch('https://api.airthreads.ai:5001/agent-rag-demo', {
          method: 'POST',
          headers: {
            'Content-Type': 'application/json',
          },
          body: JSON.stringify({
            query: input
          })
        });

        if (!response.ok) {
          throw new Error(`HTTP ${response.status}`);
        }

        // Handle SSE stream
        const reader = response.body!.getReader();
        const decoder = new TextDecoder();
        let buffer = '';

        while (true) {
          const { done, value } = await reader.read();
          if (done) break;

          buffer += decoder.decode(value, { stream: true });
          const lines = buffer.split('\n');
          buffer = lines.pop() || '';

          for (const line of lines) {
            if (line.startsWith('data: ')) {
              const data = JSON.parse(line.slice(6));

              if (data.token) {
                // Append token to streaming message
                assistantMessage.content += data.token;
                setMessages(prev => {
                  const updated = [...prev];
                  updated[updated.length - 1] = { ...assistantMessage };
                  return updated;
                });
              }
            }
            else if (line.startsWith('event: complete')) {
              // Get completion data from next line
              const nextLine = lines[lines.indexOf(line) + 1];
              if (nextLine?.startsWith('data: ')) {
                const completeData = JSON.parse(nextLine.slice(6));

                // Add meeting emails to assistant message
                if (completeData.meeting_emails?.length > 0) {
                  assistantMessage.meetingEmails = completeData.meeting_emails;
                  setMessages(prev => {
                    const updated = [...prev];
                    updated[updated.length - 1] = { ...assistantMessage };
                    return updated;
                  });
                }

                console.log('Timing:', completeData.timing);
                console.log('Metadata:', completeData.metadata);
              }
            }
            else if (line.startsWith('event: error')) {
              const nextLine = lines[lines.indexOf(line) + 1];
              if (nextLine?.startsWith('data: ')) {
                const errorData = JSON.parse(nextLine.slice(6));
                console.error('Stream error:', errorData);
                assistantMessage.content += `\n\nError: ${errorData.error}`;
                setMessages(prev => {
                  const updated = [...prev];
                  updated[updated.length - 1] = { ...assistantMessage };
                  return updated;
                });
              }
            }
          }
        }
      } catch (error) {
        console.error('Stream error:', error);
        assistantMessage.content = `Error: ${error}`;
        setMessages(prev => {
          const updated = [...prev];
          updated[updated.length - 1] = { ...assistantMessage };
          return updated;
        });
      } finally {
        setIsStreaming(false);
      }
    };

    return (
      <div className="rag-demo-chat">
        <div className="messages">
          {messages.map((msg, idx) => (
            <div key={idx} className={`message ${msg.role}`}>
              {/* Streaming text */}
              <div className="message-text">
                {msg.content}
                {isStreaming && idx === messages.length - 1 && (
                  <span className="cursor">â–‹</span>
                )}
              </div>

              {/* Meeting email cards (appear after streaming) */}
              {msg.meetingEmails && msg.meetingEmails.length > 0 && (
                <div className="email-cards">
                  {msg.meetingEmails.map((email, i) => (
                    <div key={i} className="email-card">
                      <div className="card-header">
                        <span className="event-badge">
                          {email.eventType.replace('_', ' ')}
                        </span>
                        <span className="confidence">
                          {Math.round(email.confidence * 100)}%
                        </span>
                      </div>
                      <h4 className="subject">{email.subject}</h4>
                      <p className="from">
                        From: {email.from.name || email.from.email}
                      </p>
                      <p className="date">
                        {new Date(email.date).toLocaleDateString()}
                      </p>
                      {email.snippet && (
                        <p className="snippet">{email.snippet}</p>
                      )}
                    </div>
                  ))}
                </div>
              )}
            </div>
          ))}
        </div>

        <div className="input-area">
          <input
            value={input}
            onChange={e => setInput(e.target.value)}
            onKeyPress={e => e.key === 'Enter' && sendMessage()}
            placeholder="Try: 'show me coffee chats' or 'recent meetings'"
            disabled={isStreaming}
          />
          <button onClick={sendMessage} disabled={isStreaming}>
            {isStreaming ? 'Streaming...' : 'Send'}
          </button>
        </div>
      </div>
    );
  }

  Vanilla JavaScript (Simpler)

  async function queryRAGDemo(query) {
    const response = await fetch('https://api.airthreads.ai:5001/agent-rag-demo', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ query })
    });

    const reader = response.body.getReader();
    const decoder = new TextDecoder();
    let buffer = '';
    let fullText = '';

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;

      buffer += decoder.decode(value, { stream: true });
      const lines = buffer.split('\n');
      buffer = lines.pop() || '';

      for (const line of lines) {
        if (line.startsWith('data: ')) {
          const data = JSON.parse(line.slice(6));

          if (data.token) {
            // Display token immediately
            fullText += data.token;
            document.getElementById('message').textContent = fullText;
          }
        }
        else if (line.startsWith('event: complete')) {
          const nextLine = lines[lines.indexOf(line) + 1];
          if (nextLine?.startsWith('data: ')) {
            const completeData = JSON.parse(nextLine.slice(6));

            // Render meeting cards
            if (completeData.meeting_emails?.length > 0) {
              renderMeetingCards(completeData.meeting_emails);
            }
          }
        }
      }
    }
  }

  function renderMeetingCards(emails) {
    const container = document.getElementById('email-cards');
    container.innerHTML = emails.map(email => `
      <div class="email-card">
        <span class="badge">${email.eventType}</span>
        <h4>${email.subject}</h4>
        <p>From: ${email.from.name || email.from.email}</p>
        <p>${new Date(email.date).toLocaleDateString()}</p>
      </div>
    `).join('');
  }

  Key Points

  1. Use fetch API - EventSource doesn't support POST
  2. Parse SSE manually - Split by \n, check data: and event: prefixes
  3. Stream tokens first - Append each token immediately for instant feel
  4. Cards appear after - Only render when event: complete arrives
  5. No authentication needed - This is the demo endpoint with test data
