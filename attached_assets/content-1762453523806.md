## Making voice AI simple   and accessible.

TypeScriptPythoncURLReact (web SDK)

```typescript
1# npm install @vapi-ai/server-sdk
2import { VapiClient } from '@vapi-ai/server-sdk';
3
4const vapi = new VapiClient({
5  token: 'YOUR_PRIVATE_API_KEY' // Get your private api key from the dashboard
6});
7
8async function createCall() {
9  const call = await vapi.calls.create({
10    phoneNumberId: 'YOUR_PHONE_NUMBER_ID', // Create a free phone number in the dashboard
11    customer: { number: '+1234567890' }, // Your customer's phone number
12    assistant: {
13      model: {
14        provider: 'openai',
15        model: 'gpt-4o',
16        messages: [\
17          {\
18            role: 'system',\
19            content: 'You are a helpful AI assistant. Keep your responses concise and friendly.'\
20          }\
21        ]
22      }
23    }
24  });
25}
```
